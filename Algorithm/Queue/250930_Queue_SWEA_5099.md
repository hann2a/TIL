```python
"""
deque의 popleft를 왜 쓰는지까지 이해함 
다음에는 enumerate 쓸 수 있을 것 같음
"""
import sys
from collections import deque 
sys.stdin = open('input.txt', 'r')

"""
화덕의 크기 N 피자의 개수 M 
"""

T = int(input())
for t in range(1, T+1):
    N, M = map(int, input().split())
    pizzas = deque(
        [(i+1, cheese) for i, cheese in enumerate(map(int, input().split()))]
    )

    # oven도 deque로 생성 
    oven = deque()

    # N개의 피자를 화덕에 넣음 
    for _ in range(N):
        if pizzas:
            oven.append(pizzas.popleft())
    
    last_pizza = -1 

    # 화덕이 빌 때까지 시뮬레이션 
    while oven: 
        # 입구에서 피자 꺼냄 
        pizza_num, cheese = oven.popleft()
        cheese //= 2 

        # 마지막으로 꺼낸 피자 번호 기록 
        last_pizza = pizza_num 

        # 치즈가 다 녹지 않았다면 다시 화덕에 넣음 
        if cheese > 0:
            oven.append((pizza_num, cheese))
        # 치즈가 다 녹았고, 아직 피자가 남았다면 새로운 피자를 화덕에 넣음 
        elif pizzas:
            oven.append(pizzas.popleft())
    
    print(f'#{t} {last_pizza}')
```
